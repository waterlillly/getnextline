char	*ft_strjoin(char *base, char *add)
{
	ssize_t	i;
	ssize_t	j;
	char	*join;

	i = 0;
	j = 0;
	if (add == NULL && base == NULL)
		return (NULL);
	else if (add == NULL)
		return (base);
	else if (base == NULL)
		return (add);
	join = malloc(sizeof(char) * ((ft_strlen(base) + ft_strlen(add)) + 1));
	if (!join)
		return (NULL);
	while (base[i] != '\0')
	{
		join[i] = base[i];
		i++;
	}
	while (add[j] != '\0')
		join[i++] = add[j++];
	join[i] = '\0';
	return (join);
}
